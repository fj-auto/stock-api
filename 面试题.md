设计一个算法，计算停车场的收费金额。停车场收费标准如下：
1、前 2 小时收费 5 元每小时。
2、超过 2 小时后，每小时收费 3 元。
3、不足 1 小时的部分按 1 小时计费。
4、若停车时间超过 8 小时，按全天收费 30 元封顶。
5、每周日全天免费停车。

```js
function calculateParkingFee(minutes, dateStr) {
  const date = new Date(dateStr);
  const isSunday = date.getDay() === 0;

  if (isSunday) {
    return 0; // 周日免费停车
  }

  const hours = Math.ceil(minutes / 60);

  if (hours > 8) {
    return 30; // 全天封顶收费
  } else if (hours <= 2) {
    return hours * 5;
  } else {
    return 2 * 5 + (hours - 2) * 3;
  }
}
```

公交公司希望推出一种组合车票，每张组合车票允许乘客选择线路中任意三个不同站台乘车一次，且这三个站台收费金额相加后，刚好等于一个预先设定好的总金额（例如 50 元）。公司需要找出所有满足这个金额条件的站台组合，以便进行组合车票的设计与销售。

```js
// 组合车票计算函数（前述实现）
function findStationCombinations(stations, target) {
  // 创建空数组存储最终结果，每个结果是一个包含三个站台费用的数组
  const results = [];

  // 对站台费用进行升序排序，是双指针算法的必要前提
  // 排序后可以根据总和与目标值的比较，决定移动哪个指针
  stations.sort((a, b) => a - b);

  // 外层循环固定第一个站台，最多到倒数第三个元素
  // 因为需要为另外两个站台预留至少两个位置
  for (let i = 0; i < stations.length - 2; i++) {
    // 初始化左指针为当前站台的下一个位置
    let left = i + 1;
    // 初始化右指针为数组的最后一个元素
    // 这样左右指针之间的所有元素都是可能的候选站台
    let right = stations.length - 1;

    // 双指针向中间移动，直到相遇为止
    // 双指针技术能将算法复杂度从O(n³)降到O(n²)
    while (left < right) {
      // 计算当前三个站台的费用总和
      const total = stations[i] + stations[left] + stations[right];

      if (total === target) {
        // 找到符合条件的组合，将其添加到结果数组
        // 这个组合的三个站台费用之和刚好等于目标金额
        results.push([stations[i], stations[left], stations[right]]);

        // 同时移动左右指针寻找其他可能的组合
        // 这样可以避免重复，并确保找到所有可能的解
        left++;
        right--;
      } else if (total < target) {
        // 总和小于目标值，需要增加总和
        // 由于数组已排序，移动左指针可以获取更大的值
        left++;
      } else {
        // 总和大于目标值，需要减少总和
        // 由于数组已排序，移动右指针可以获取更小的值
        right--;
      }
    }
  }

  // 返回所有满足条件的站台组合
  // 这些组合将用于公交公司设计组合车票
  return results;
}
```

假设你今天安排了很多活动，每个活动都有开始和结束的时间。例如，你上午有一个活动从 9:00 开始到 10:00 结束，又有一个活动从 9:30 开始到 10:30 结束。由于这些活动有部分时间重叠，为了更清楚地知道你什么时候在忙，你希望把所有重叠的时间段合并成连续的时间段。请你帮忙整理出一个不重叠的时间段列表，使得这些时间段能恰好覆盖你原本所有活动的时间。

```js
function mergeIntervals(intervals) {
  // 如果活动列表为空，则直接返回空数组
  if (!intervals || intervals.length === 0) return [];

  // 按照每个活动的开始时间进行排序
  intervals.sort((a, b) => a[0] - b[0]);

  // 初始化合并后的活动数组，将第一个活动加入结果
  let merged = [intervals[0]];

  // 遍历从第二个开始的所有活动，判断是否存在重叠
  for (let i = 1; i < intervals.length; i++) {
    let current = intervals[i]; // 当前活动的时间段
    let lastMerged = merged[merged.length - 1]; // 合并后结果中的最后一个活动

    // 如果当前活动与最后一个合并活动存在重叠或边界相连，则合并它们
    if (current[0] <= lastMerged[1]) {
      lastMerged[1] = Math.max(lastMerged[1], current[1]);
    } else {
      // 如果不重叠，则直接加入当前活动
      merged.push(current);
    }
  }

  // 返回合并后的活动时间段
  return merged;
}
```

```html
            if (!sidebar.classList.contains('collapsed')) {
              wasCollapsedBeforeResize = false;
            }
            sidebar.classList.remove('collapsed');

            // 默认在移动端自动折叠（不显示expanded-mobile类）
            sidebar.classList.remove('expanded-mobile');

            // 确保内容区域有正确的margin
            content.style.marginLeft = '60px';
            content.style.width = 'calc(100% - 60px)';
          } else {
            // 非移动设备宽度，恢复到调整大小前的状态
            sidebar.classList.remove('expanded-mobile');
            content.style.marginLeft = '';
            content.style.width = '';

            if (wasCollapsedBeforeResize) {
              sidebar.classList.add('collapsed');
            } else {
              sidebar.classList.remove('collapsed');
            }
          }
        });

        // 页面加载时检查屏幕宽度
        if (window.innerWidth <= 600) {
          sidebar.classList.remove('collapsed');
          sidebar.classList.remove('expanded-mobile');
          // 确保内容区域有正确的margin
          content.style.marginLeft = '60px';
          content.style.width = 'calc(100% - 60px)';
        }
      });
```

```js
function findStationCombinations(stations, target) {
  const result = [];
  stations.sort((a, b) => a - b);

  for (let i = 0; i < stations.length - 2; i++) {
    if (i > 0 && stations[i] === stations[i - 1]) {
      continue;
    }

    let left = i + 1;
    let right = stations.length - 1;
    while (left < right) {
      const sum = stations[i] + stations[left] + stations[right];

      if (sum === target) {
        result.push(stations[i], stations[left], stations[right]);

        while (left < right && stations[left] === stations[left + 1]) {
          left++;
        }
        while (left < right && stations[right] === stations[right - 1]) {
          right--;
        }
      } else if (sum < target) {
        left++;
      } else {
        right--;
      }
    }
  }
}

console.log('测试用例1:', findStationCombinations([5, 10, 15, 20, 25, 30], 50)); // 测试用例1: [ [ 5, 15, 30 ], [ 5, 20, 25 ], [ 10, 15, 25 ] ]
console.log('测试用例2:', findStationCombinations([4, 8, 12, 16, 20], 100)); // 测试用例2: []
```

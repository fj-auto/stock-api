function findStationCombinations(stations, target) {
  const result = [];
  stations.sort((a, b) => a - b);

  for (let i = 0; i < stations.length - 2; i++) {
    if (i > 0 && stations[i] === stations[i - 1]) {
      continue;
    }

    let left = i + 1;
    let right = stations.length - 1;
    while (left < right) {
      const sum = stations[i] + stations[left] + stations[right];

      if (sum === target) {
        result.push(stations[i], stations[left], stations[right]);

        while (left < right && stations[left] === stations[left + 1]) {
          left++;
        }
        while (left < right && stations[right] === stations[right - 1]) {
          right--;
        }
      } else if (sum < target) {
        left++;
      } else {
        right--;
      }
    }
  }
}
